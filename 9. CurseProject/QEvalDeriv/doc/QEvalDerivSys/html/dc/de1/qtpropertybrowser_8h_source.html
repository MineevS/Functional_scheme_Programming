112template <class PropertyManager>
113class QtAbstractEditorFactory : public QtAbstractEditorFactoryBase
114{
115public:
116 explicit QtAbstractEditorFactory(QObject *parent) : QtAbstractEditorFactoryBase(parent) {}
117 QWidget *createEditor(QtProperty *property, QWidget *parent) override
118 {
119 for (PropertyManager *manager : m_managers) { //std::as_const(m_managers)
120 if (manager == property->propertyManager()) {
121 return createEditor(manager, property, parent);
122 }
123 }
124 return 0;
125 }
126 void addPropertyManager(PropertyManager *manager)
127 {
128 if (m_managers.contains(manager))
129 return;
130 m_managers.insert(manager);
131 connectPropertyManager(manager);
132 connect(manager, &QObject::destroyed,
133 this, &QtAbstractEditorFactory<PropertyManager>::managerDestroyed);
134 }
135 void removePropertyManager(PropertyManager *manager)
136 {
137 if (!m_managers.contains(manager))
138 return;
139 disconnect(manager, &QObject::destroyed,
140 this, &QtAbstractEditorFactory<PropertyManager>::managerDestroyed);
141 disconnectPropertyManager(manager);
142 m_managers.remove(manager);
143 }
144 QSet<PropertyManager *> propertyManagers() const
145 {
146 return m_managers;
147 }
148 PropertyManager *propertyManager(QtProperty *property) const
149 {
150 QtAbstractPropertyManager *manager = property->propertyManager();
151 for (PropertyManager *m : m_managers) { // std::as_const(m_managers)
152 if (m == manager) {
153 return m;
154 }
155 }
156 return 0;
157 }
158protected:
159 virtual void connectPropertyManager(PropertyManager *manager) = 0;
160 virtual QWidget *createEditor(PropertyManager *manager, QtProperty *property,
161 QWidget *parent) = 0;
162 virtual void disconnectPropertyManager(PropertyManager *manager) = 0;
163 void managerDestroyed(QObject *manager) override
164 {
165 for (PropertyManager *m : m_managers) { // std::as_const(m_managers)
166 if (m == manager) {
167 m_managers.remove(m);
168 return;
169 }
170 }
171 }
172private:
173 void breakConnection(QtAbstractPropertyManager *manager) override
174 {
175 for (PropertyManager *m : m_managers) { // std::as_const(m_managers)
176 if (m == manager) {
177 removePropertyManager(m);
178 return;
179 }
180 }
181 }
182private:
183 QSet<PropertyManager *> m_managers;
184 friend class QtAbstractPropertyEditor;
185};
